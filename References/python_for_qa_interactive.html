while loop — retry logic
max_retries = 3 attempt = 0 success = False while attempt < max_retries and not success: attempt += 1 print(f"Attempt {attempt}...") # try the action here if attempt == 2: # simulate success on attempt 2 success = True print("Succeeded!") if not success: print("Failed after all retries")
The for item in list pattern is how pytest parametrize works under the hood. When you write @pytest.mark.parametrize, pytest runs your test function once for each item — exactly like a for loop.

def keyword
Defines a reusable block of code. Give it a clear verb name: login(), get_error(), is_visible()
Always use
return
Sends a value back to the caller. Functions without return give back None
For helpers
default args
Set defaults with param=value. Callers can override — perfect for happy vs negative path
Very useful
docstring
The triple-quoted string right after def. Explains what the function does. Write these always.
Good habit
If you find yourself copying and pasting the same 3 lines in multiple tests — stop and put them in a function. That's the single most important coding habit for QA automation.
Why OOP matters for QA — the Page Object Model
The #1 use of OOP in test automation is the Page Object Model (POM). You create one class per page of your app. The class holds all locators and actions for that page. Tests just call methods — they never touch raw selectors.
Step 1 — a class is a blueprint
# Class = blueprint. Instance = the actual object you use. class LoginPage: # __init__ runs when you create an instance def __init__(self, page): self.page = page # store the page object # Method = a function that belongs to this class def navigate(self): self.page.goto("https://the-internet.herokuapp.com/login") def login(self, username, password): self.page.get_by_label("Username").fill(username) self.page.get_by_label("Password").fill(password) self.page.get_by_role("button", name="Login").click() def get_error_message(self): return self.page.locator("#flash.error").inner_text().strip()
Step 2 — use the class in your tests
# test_login.py from pages.login_page import LoginPage from playwright.sync_api import Page, expect def test_valid_login(page: Page): login = LoginPage(page) # create an instance login.navigate() login.login("tomsmith", "SuperSecretPassword!") expect(page.get_by_text("You logged into a secure area!")).to_be_visible() def test_invalid_login(page: Page): login = LoginPage(page) login.navigate() login.login("baduser", "badpass") assert "invalid" in login.get_error_message().lower()
Step 3 — inheritance (when pages share behaviour)
# Base class — things every page has class BasePage: def __init__(self, page): self.page = page def wait_for_spinner(self): self.page.locator(".loading-spinner").wait_for(state="hidden") def get_page_title(self): return self.page.title() # Child class inherits everything from BasePage class LoginPage(BasePage): # ← inherits from BasePage def login(self, username, password): self.page.get_by_label("Username").fill(username) self.page.get_by_label("Password").fill(password) self.page.get_by_role("button", name="Login").click() self.wait_for_spinner() # ← inherited from BasePage! # Now every page class you write gets wait_for_spinner() for free
self
Refers to the current instance. Think of it as "this object's own data". Always the first parameter of any method.
Essential
__init__
The constructor — runs automatically when you create an instance with LoginPage(page)
Essential
inheritance
class Child(Parent) — Child gets all Parent methods. Use for a BasePage every page inherits from.
POM pattern
instance
The actual object you create: login = LoginPage(page). You can create many instances from one class.

Exception handling — graceful test failures
from playwright.sync_api import TimeoutError as PlaywrightTimeout def safe_click(page, selector): """Click element, skip gracefully if not found.""" try: page.click(selector, timeout=3000) print(f"Clicked: {selector}") except PlaywrightTimeout: print(f"Element not found: {selector} — skipping") # Useful for: cookie banners, popups, optional UI elements safe_click(page, "#cookie-accept") # won't crash if no cookie banner
