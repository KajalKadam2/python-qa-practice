Day 18 — Locators in depth
A locator tells Playwright which element on the page to interact with. Choosing the right locator determines whether your tests are stable or flaky. There is a clear priority order — use the highest-ranked locator that works for your element.
The locator priority order — memorise this
1get_by_role()Accessibility role — best for buttons, links, headings, inputs
2get_by_label()Form label — best for all form fields
3get_by_placeholder()Placeholder text — when no label exists
4get_by_text()Visible text — for links and text content
5get_by_test_id()data-testid attribute — when devs add test IDs
6locator("css")CSS selector — when semantic locators do not work
7locator("xpath")XPath — last resort only
Playwright's own documentation recommends this exact priority. Semantic locators (1–5) survive UI redesigns because they describe what the element is, not where it is on the page. CSS and XPath break when developers move things around.
1 of 9
get_by_role() — the most powerful locator use first always
get_by_role() finds elements by their ARIA role — the accessibility label that describes what the element is. A button is a button, a heading is a heading, a link is a link. This matches how screen readers see the page and survives CSS class changes.
Common roles you will use every day
Type this — test_locators_role.py
from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" def test_role_button(page: Page): page.goto(f"{BASE}/login") # Button by role + name btn = page.get_by_role("button", name="Login") expect(btn).to_be_visible() expect(btn).to_be_enabled() def test_role_heading(page: Page): page.goto(f"{BASE}/login") # Heading by role + exact name heading = page.get_by_role("heading", name="Login Page") expect(heading).to_be_visible() def test_role_link(page: Page): page.goto(f"{BASE}") # Link by role + name link = page.get_by_role("link", name="Form Authentication") expect(link).to_be_visible() link.click() expect(page).to_have_url(f"{BASE}/login") def test_role_checkbox(page: Page): page.goto(f"{BASE}/checkboxes") # Checkbox by role checkboxes = page.get_by_role("checkbox") expect(checkboxes).to_have_count(2) def test_role_exact(page: Page): page.goto(f"{BASE}/secure" if False else f"{BASE}/login") # exact=True — matches ONLY "Login" not "Login Page" btn = page.get_by_role("button", name="Login", exact=True) expect(btn).to_be_visible()
All roles you will encounter
# Most common ARIA roles in web apps "button" # <button> or role="button" "link" # <a href> "heading" # <h1> through <h6> "textbox" # <input type="text"> "checkbox" # <input type="checkbox"> "radio" # <input type="radio"> "combobox" # <select> dropdown "listitem" # <li> in a list "alert" # role="alert" — flash messages "img" # <img> with alt text "navigation" # <nav> "main" # <main>
2 of 9
get_by_label, get_by_text, get_by_placeholder
These three cover the remaining semantic locators. get_by_label is for form fields. get_by_text is for visible text. get_by_placeholder is a fallback when no label exists.
Type this
from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" # ── get_by_label ───────────────────────────────── def test_label_locator(page: Page): page.goto(f"{BASE}/login") # Finds the input associated with the label "Username" page.get_by_label("Username").fill("tomsmith") page.get_by_label("Password").fill("SuperSecretPassword!") # Verify values were filled expect(page.get_by_label("Username")).to_have_value("tomsmith") # ── get_by_text ─────────────────────────────────── def test_text_locator(page: Page): page.goto(f"{BASE}/login") page.get_by_label("Username").fill("tomsmith") page.get_by_label("Password").fill("SuperSecretPassword!") page.get_by_role("button", name="Login").click() # get_by_text — finds element containing this text expect(page.get_by_text("You logged into a secure area!")).to_be_visible() # exact=True — must match exactly, no partial matches expect(page.get_by_text("Secure Area", exact=True)).to_be_visible() # ── get_by_placeholder ─────────────────────────── def test_placeholder_locator(page: Page): # Use when no label exists — placeholder is fallback page.goto("https://the-internet.herokuapp.com/login") # If the input had placeholder="Enter your username" # page.get_by_placeholder("Enter your username").fill("tomsmith") # Our login page uses labels so use get_by_label instead page.get_by_label("Username").fill("tomsmith") expect(page.get_by_label("Username")).to_have_value("tomsmith")
When get_by_text("Secure Area") matches multiple elements you get a strict mode violation. Fix with exact=True or switch to get_by_role("heading", name="Secure Area", exact=True). You solved this exact error in Day 16.
3 of 9
CSS selectors — when semantic locators are not enough rank 6
CSS selectors target elements by their HTML attributes — ID, class, tag, attribute value. Use them when an element has no accessible role, no label, and no visible text. They are more fragile than semantic locators but more readable than XPath.
HTML we are targeting
<div id="flash" class="flash success"> You logged into a secure area! </div> <input type="checkbox" checked> <select id="dropdown"></select> <button class="btn-danger">Delete</button>
Type this — CSS selector patterns
from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" def test_css_by_id(page: Page): page.goto(f"{BASE}/login") page.get_by_label("Username").fill("tomsmith") page.get_by_label("Password").fill("SuperSecretPassword!") page.get_by_role("button", name="Login").click() # #id selector expect(page.locator("#flash")).to_be_visible() def test_css_by_class(page: Page): page.goto(f"{BASE}/login") page.get_by_label("Username").fill("wronguser") page.get_by_label("Password").fill("wrongpass") page.get_by_role("button", name="Login").click() # .class selector — element with BOTH flash and error classes expect(page.locator("#flash.error")).to_contain_text("invalid") def test_css_by_attribute(page: Page): page.goto(f"{BASE}/checkboxes") # [attr] — element with this attribute checked = page.locator("input[type='checkbox'][checked]") expect(checked).to_be_checked() def test_css_by_tag(page: Page): page.goto(f"{BASE}/add_remove_elements/") page.get_by_role("button", name="Add Element").click() page.get_by_role("button", name="Add Element").click() # tag + class selector expect(page.locator("button.added-manually")).to_have_count(2)
CSS selector quick reference
"#flash" # by ID ".error" # by class "#flash.error" # ID + class combined "input[type='checkbox']" # by attribute "button.added-manually" # tag + class "div > p" # direct child "ul li:first-child" # first list item "ul li:nth-child(2)" # second list item
4 of 9
XPath — last resort only rank 7
XPath traverses the entire HTML tree to find elements. It is the most powerful locator — it can find things nothing else can — but also the most brittle. Any change in page structure breaks XPath. Use it only when all other locators fail.
Type this — XPath patterns you actually need
from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" def test_xpath_by_text(page: Page): page.goto(f"{BASE}/login") # XPath to find element containing specific text label = page.locator("//label[contains(text(),'Username')]") expect(label).to_be_visible() def test_xpath_by_attribute(page: Page): page.goto(f"{BASE}/login") # XPath by attribute value username = page.locator("//input[@id='username']") username.fill("tomsmith") expect(username).to_have_value("tomsmith") def test_xpath_parent_child(page: Page): page.goto(f"{BASE}/checkboxes") # XPath: find form, then get checkboxes inside it checkboxes = page.locator("//form[@id='checkboxes']//input") expect(checkboxes).to_have_count(2)
XPath quick reference
"//div" # any div anywhere "//input[@id='username']" # by attribute "//label[text()='Username']" # exact text match "//label[contains(.,'User')]" # partial text match "//form//input" # input inside a form "//ul/li[1]" # first li (XPath is 1-indexed) "//button[not(@disabled)]" # button that is not disabled
XPath is 1-indexed — //li[1] is the first item. CSS :nth-child() is also 1-indexed. But Python lists and Playwright's .nth() are 0-indexed. Do not mix them up.

nth, filter, and chaining — advanced locator patterns interview level
When a locator matches multiple elements you need to narrow it down. .nth() picks by position, .filter() picks by content or state, and chaining combines locators for precision.
Type this
from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" # ── .nth() — pick by position ─────────────────── def test_nth(page: Page): page.goto(f"{BASE}/checkboxes") checkboxes = page.locator("input[type='checkbox']") first = checkboxes.nth(0) # 0-indexed second = checkboxes.nth(1) last = checkboxes.last # shortcut for last first_item = checkboxes.first # shortcut for first expect(first).not_to_be_checked() expect(second).to_be_checked() # ── .filter() — narrow by content or state ────── def test_filter(page: Page): page.goto(f"{BASE}/add_remove_elements/") for _ in range(3): page.get_by_role("button", name="Add Element").click() # Filter buttons to only "Delete" ones delete_buttons = page.get_by_role("button").filter(has_text="Delete") expect(delete_buttons).to_have_count(3) # ── Chaining — narrow from parent to child ─────── def test_chain(page: Page): page.goto(f"{BASE}/login") # Find the form, then find the button inside it form = page.locator("#login") submit = form.get_by_role("button") expect(submit).to_be_visible() # ── .first and .last shortcuts ─────────────────── def test_first_last(page: Page): page.goto(f"{BASE}/add_remove_elements/") for _ in range(2): page.get_by_role("button", name="Add Element").click() deletes = page.get_by_role("button", name="Delete") # Click only the first Delete button deletes.first.click() expect(deletes).to_have_count(1)
Chaining is the cleanest way to handle ambiguous locators. Instead of a complex CSS or XPath that reaches deep into the DOM, find the parent container first, then find the child element within it. Each step is simple and readable.

Debugging locators — finding the right selector
When a locator fails, you need to investigate. These tools let you find the right locator without trial and error.
Tool 1 — Playwright Inspector (best tool)
# Run with --headed and pause to inspect pytest test_login_pom.py -v --headed --slowmo=1000 # Or add a pause inside your test def test_debug(page: Page): page.goto("https://the-internet.herokuapp.com/login") page.pause() # opens Playwright Inspector — hover to highlight elements
Tool 2 — Browser DevTools
# In any browser — right-click element → Inspect # In the Console tab type: document.querySelector("#username") # test CSS selector document.querySelectorAll(".flash") # see how many match $x("//input[@id='username']") # test XPath
Tool 3 — Playwright codegen (auto-generates locators)

page.get_by_role("button", name="Login").click()
When you get strict mode violation
# Error: strict mode violation: get_by_text("X") resolved to N elements # Fix options — in priority order: page.get_by_role("heading", name="X", exact=True) # most specific page.get_by_text("X", exact=True) # exact match page.locator("h2").filter(has_text="X") # filter by tag + text page.get_by_text("X").first # take first match

# test_day18_locators.py — locator showcase import pytest from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" def test_semantic_locators(page: Page): """Demonstrate get_by_role and get_by_label.""" page.goto(f"{BASE}/login") # role: heading expect(page.get_by_role("heading", name="Login Page")).to_be_visible() # label: form fields page.get_by_label("Username").fill("tomsmith") page.get_by_label("Password").fill("SuperSecretPassword!") # role: button page.get_by_role("button", name="Login").click() # text: flash message expect(page.get_by_text("You logged into a secure area!")).to_be_visible() # role: heading exact — avoids strict mode violation expect(page.get_by_role("heading", name="Secure Area", exact=True)).to_be_visible() def test_css_locators(page: Page): """Demonstrate CSS selectors.""" page.goto(f"{BASE}/login") page.get_by_label("Username").fill("wronguser") page.get_by_label("Password").fill("wrongpass") page.get_by_role("button", name="Login").click() # CSS: ID + class expect(page.locator("#flash.error")).to_contain_text("Your username is invalid") def test_nth_and_filter(page: Page): """Demonstrate .nth() and .filter().""" page.goto(f"{BASE}/checkboxes") # nth(0) = first, nth(1) = second expect(page.locator("input[type='checkbox']").nth(0)).not_to_be_checked() expect(page.locator("input[type='checkbox']").nth(1)).to_be_checked() def test_dynamic_elements(page: Page): """Demonstrate to_have_count on dynamic list.""" page.goto(f"{BASE}/add_remove_elements/") for _ in range(3): page.get_by_role("button", name="Add Element").click() delete_btns = page.get_by_role("button").filter(has_text="Delete") expect(delete_btns).to_have_count(3) delete_btns.first.click() expect(delete_btns).to_have_count(2)
