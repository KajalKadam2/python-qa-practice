Day 18 — Locators in depth
A locator is how Playwright finds an element on the page. Pick the wrong locator strategy and your tests become brittle — they break every time a developer changes a class name or moves an element. Pick the right one and tests survive UI changes for months.

Today you learn all 7 locator types, when to use each, and how to handle the tricky cases — multiple matches, nested elements, dynamic content.
The 7 locator types — priority order
#1
get_by_role()
Finds by ARIA role — button, link, heading, textbox, checkbox. Most resilient — tied to semantics not styling.
#2
get_by_label()
Finds form inputs by their label text. Perfect for login forms, search boxes, any labeled input.
#3
get_by_placeholder()
Finds inputs by placeholder text. Good fallback when no label exists.
#4
get_by_text()
Finds elements by visible text content. Good for links and buttons with text.
#5
get_by_test_id()
Finds by data-testid attribute. Ideal when dev team adds test IDs — most stable of all.
#6
CSS selector
locator("#id"), locator(".class"), locator("input[type='checkbox']"). Use when semantic locators don't work.
#7
XPath
locator("//button[@type='submit']"). Last resort — fragile, verbose, but powerful for complex cases.
Always start at #1 and work down. If get_by_role works — use it and stop. Only move down the list when the higher option does not work or produces strict mode violations.
1 of 9
get_by_role() — the locator you use 80% of the time
ARIA roles describe what an element IS — a button, a link, a heading, a checkbox. They survive CSS class changes and ID changes. This is why get_by_role is always first choice.
Common roles you will use every day
# Buttons page.get_by_role("button", name="Login") page.get_by_role("button", name="Submit") page.get_by_role("button", name="Add Element") # Links page.get_by_role("link", name="Logout") page.get_by_role("link", name="Home") # Headings page.get_by_role("heading", name="Secure Area", exact=True) page.get_by_role("heading", level=1) # h1 specifically # Form inputs page.get_by_role("textbox", name="Username") page.get_by_role("checkbox") page.get_by_role("combobox") # dropdown select
Create Day18_locators.py — type this first
from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" def test_get_by_role_button(page: Page): page.goto(f"{BASE}/login") # Find the login button by its role and name btn = page.get_by_role("button", name="Login") expect(btn).to_be_visible() expect(btn).to_be_enabled() def test_get_by_role_heading(page: Page): page.goto(f"{BASE}/login") # exact=True prevents matching substrings heading = page.get_by_role("heading", name="Login Page", exact=True) expect(heading).to_be_visible() def test_get_by_role_link(page: Page): page.goto(f"{BASE}") # Find a link by its text link = page.get_by_role("link", name="Form Authentication") expect(link).to_be_visible() link.click() expect(page).to_have_url(f"{BASE}/login")
pytest Day18_locators.py -v --headed
exact=True means the name must match exactly — not just contain the text. Always use it for headings and links to avoid strict mode violations where multiple elements match.
2 of 9
get_by_label, get_by_placeholder, get_by_text
These three cover form inputs and content. get_by_label is the best for any input with a visible label. get_by_placeholder is the fallback when no label exists. get_by_text finds elements by their visible text — but watch out for strict mode violations.
Add these to Day18_locators.py
# ── get_by_label ───────────────────────────────── def test_get_by_label(page: Page): page.goto(f"{BASE}/login") # Label "Username" is linked to the input via for= attribute page.get_by_label("Username").fill("tomsmith") page.get_by_label("Password").fill("SuperSecretPassword!") expect(page.get_by_label("Username")).to_have_value("tomsmith") # ── get_by_placeholder ─────────────────────────── def test_get_by_placeholder(page: Page): # inputs page has fields with placeholder text page.goto(f"{BASE}/inputs") number_input = page.get_by_role("spinbutton") number_input.fill("42") expect(number_input).to_have_value("42") # ── get_by_text ────────────────────────────────── def test_get_by_text(page: Page): page.goto(f"{BASE}/login") # exact=True prevents strict mode violations expect( page.get_by_text("Login Page", exact=True) ).to_be_visible() def test_get_by_text_link(page: Page): page.goto(f"{BASE}") # Click a link by its text page.get_by_text("Checkboxes").click() expect(page).to_have_url(f"{BASE}/checkboxes")
When get_by_text("Secure Area") matches multiple elements — you saw this error on Day 16 — add exact=True or switch to get_by_role("heading", name="Secure Area", exact=True). This is the most common locator mistake.

CSS selectors — locator() for precise targeting
CSS selectors use the same syntax as CSS stylesheets. They are more powerful than semantic locators for targeting specific element types, attributes, or positions. Use them when no semantic locator works cleanly.
CSS selector patterns you need to know
# By ID — the most specific CSS selector page.locator("#username") # element with id="username" page.locator("#flash") # the flash message div # By class page.locator(".flash") # any element with class="flash" page.locator(".flash.success") # has BOTH classes page.locator(".flash.error") # has BOTH classes # By element type page.locator("button") # all buttons page.locator("h2") # all h2 headings # By attribute page.locator("input[type='checkbox']") # checkboxes page.locator("input[type='text']") # text inputs page.locator("button[type='submit']") # submit button page.locator("a[href='/login']") # link with specific href # Combined page.locator("div#flash.success") # div with id=flash AND class=success
Add to Day18_locators.py
def test_css_id_selector(page: Page): page.goto(f"{BASE}/login") # Use CSS ID selector for the username input page.locator("#username").fill("tomsmith") page.locator("#password").fill("SuperSecretPassword!") page.locator("button[type='submit']").click() expect(page.locator("#flash.success")).to_be_visible() def test_css_attribute_selector(page: Page): page.goto(f"{BASE}/checkboxes") # Target checkboxes by type attribute checkboxes = page.locator("input[type='checkbox']") expect(checkboxes).to_have_count(2) def test_css_class_selector(page: Page): page.goto(f"{BASE}/login") page.locator("#username").fill("wrong") page.locator("#password").fill("wrong") page.locator("button[type='submit']").click() # Flash error has both .flash and .error classes expect(page.locator("#flash.error")).to_be_visible()
CSS selectors like #flash.error are how you handled flash messages in Day 16 and 17. You already know them — this page formalises the patterns.
4 of 9
XPath — last resort but powerful use sparingly
XPath is a query language for navigating XML/HTML. It can do things CSS cannot — traverse upward to parents, find elements by text content, select by position. Use it only when CSS and semantic locators fail.
XPath patterns you need to know
# Basic syntax — always starts with // page.locator("//button") # all buttons page.locator("//button[@type='submit']") # button with attribute page.locator("//button[text()='Login']") # button with exact text page.locator("//button[contains(text(),'Log')]") # button containing text # By ID and class page.locator("//*[@id='username']") # any element with id page.locator("//*[@class='flash success']") # exact class match # Traversing relationships page.locator("//form//input[@type='text']") # input inside a form page.locator("//label[text()='Username']/../input") # sibling of label # By position page.locator("(//input[@type='checkbox'])[1]") # first checkbox page.locator("(//input[@type='checkbox'])[2]") # second checkbox
Add to Day18_locators.py
def test_xpath_button(page: Page): page.goto(f"{BASE}/login") page.locator("//*[@id='username']").fill("tomsmith") page.locator("//*[@id='password']").fill("SuperSecretPassword!") page.locator("//button[@type='submit']").click() expect(page.locator("//div[@id='flash']")).to_contain_text("secure area") def test_xpath_checkboxes_by_position(page: Page): page.goto(f"{BASE}/checkboxes") # XPath position() starts at 1 not 0 first = page.locator("(//input[@type='checkbox'])[1]") second = page.locator("(//input[@type='checkbox'])[2]") expect(first).not_to_be_checked() expect(second).to_be_checked()
XPath tests break easily when developers restructure the HTML — adding a wrapper div or moving an element changes the path. Always prefer get_by_role or CSS. Only use XPath when absolutely nothing else works.

Handling multiple matches — .nth() .first .last .filter()
When a locator matches multiple elements, Playwright's strict mode throws an error. These four methods let you pick exactly which element you want from a list of matches.
All four methods
# .nth(index) — pick by position (0-based) page.locator("input[type='checkbox']").nth(0) # first page.locator("input[type='checkbox']").nth(1) # second # .first and .last — shorthand page.locator("input[type='checkbox']").first # same as .nth(0) page.locator("input[type='checkbox']").last # last one # .filter(has_text=) — narrow by text content page.locator("li").filter(has_text="Option 1") page.locator(".menu-item").filter(has_text="Settings") # Chaining — find parent then child page.locator("form").locator("input[type='text']")
Add to Day18_locators.py
def test_nth_checkbox(page: Page): page.goto(f"{BASE}/checkboxes") checkboxes = page.locator("input[type='checkbox']") # Total count expect(checkboxes).to_have_count(2) # First is unchecked, second is checked expect(checkboxes.nth(0)).not_to_be_checked() expect(checkboxes.nth(1)).to_be_checked() # .first and .last are shorthand expect(checkboxes.first).not_to_be_checked() expect(checkboxes.last).to_be_checked() def test_chained_locator(page: Page): page.goto(f"{BASE}/login") # Find the form first, then inputs inside it form = page.locator("form") form.locator("#username").fill("tomsmith") form.locator("#password").fill("SuperSecretPassword!") form.get_by_role("button", name="Login").click() expect(page.locator("#flash.success")).to_be_visible()
.nth() is 0-based like Python lists. XPath position() is 1-based. This trips people up — .nth(0) is the first element, (//input)[1] is also the first element.

playwright codegen — auto-generate locators saves hours
Playwright's codegen tool opens a browser and records your clicks — generating Python code with locators automatically. Instead of inspecting HTML and writing locators manually, you click through the app and Playwright writes the code for you.
How to use codegen
# Run codegen pointing at your test site playwright codegen https://the-internet.herokuapp.com/login
What happens
# Two windows open: # 1. A browser — interact with it normally # 2. A code panel — shows generated code as you click # Example: you click Username field and type "hello" # Code panel shows: page.get_by_label("Username").fill("hello") # You click the Login button # Code panel shows: page.get_by_role("button", name="Login").click()
Run it now
# In terminal: playwright codegen https://the-internet.herokuapp.com # Click through these pages and watch the code generate: # 1. Click "Form Authentication" → login page # 2. Fill username and password # 3. Click Login button # 4. Click Logout # Copy the generated code into a new file
Use codegen to debug locators too
# Add page.pause() to stop a running test and inspect the page def test_debug_locator(page: Page): page.goto(f"{BASE}/checkboxes") page.pause() # browser freezes here — opens inspector # hover over elements in the browser to see locator suggestions # press Resume when done
Use codegen at the start of every new test page. It gives you the best available locators in seconds. Then review what it generates — if it uses XPath, see if you can rewrite it with get_by_role instead.

This is the reference file for Day 18. Every locator type, every method. Commit this to your qa-automation repo as your locator reference.
Create Day18_all_locators.py
# Day18_all_locators.py — complete locator reference import pytest from playwright.sync_api import Page, expect BASE = "https://the-internet.herokuapp.com" # ── Role locators ───────────────────────────────── def test_role_locators(page: Page): page.goto(f"{BASE}/login") expect(page.get_by_role("button", name="Login")).to_be_visible() expect(page.get_by_role("heading", name="Login Page", exact=True)).to_be_visible() # ── Label locators ──────────────────────────────── def test_label_locators(page: Page): page.goto(f"{BASE}/login") page.get_by_label("Username").fill("tomsmith") page.get_by_label("Password").fill("SuperSecretPassword!") page.get_by_role("button", name="Login").click() expect(page.locator("#flash.success")).to_be_visible() # ── CSS locators ────────────────────────────────── def test_css_locators(page: Page): page.goto(f"{BASE}/checkboxes") expect(page.locator("input[type='checkbox']")).to_have_count(2) expect(page.locator("input[type='checkbox']").first).not_to_be_checked() expect(page.locator("input[type='checkbox']").last).to_be_checked() # ── XPath locators ──────────────────────────────── def test_xpath_locators(page: Page): page.goto(f"{BASE}/login") page.locator("//*[@id='username']").fill("tomsmith") page.locator("//*[@id='password']").fill("SuperSecretPassword!") page.locator("//button[@type='submit']").click() expect(page.locator("//div[@id='flash']")).to_contain_text("secure area") # ── nth, first, last ────────────────────────────── def test_nth_locators(page: Page): page.goto(f"{BASE}/checkboxes") page.locator("input[type='checkbox']").nth(0).check() expect(page.locator("input[type='checkbox']").nth(0)).to_be_checked() # ── Dropdown with CSS ───────────────────────────── def test_dropdown_css(page: Page): page.goto(f"{BASE}/dropdown") page.locator("#dropdown").select_option("1") expect(page.locator("#dropdown")).to_have_value("1")
pytest Day18_all_locators.py -v --headed
